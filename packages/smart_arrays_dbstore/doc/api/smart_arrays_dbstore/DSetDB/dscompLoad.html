dscompLoad method

Future<List<Float64List>> dscompLoad (DSKey dskey, String dsc)
override

Inverse of dscompSave: Returns the matrix that was stored as dataset component dsc in the database whose name is derived from dskey. Returns null if the component doesn't exist, or an error occured. Handles both cases: Data are stored in a single block or in several blocks by dscompSave.

Implementation

Future<List<Float64List>> dscompLoad(DSKey dskey, String dsc) async {
  String dbname = dbnameGet(dskey);
  if (dbname == null) return new Future.value(null); // dataset does not exist

  Database db;
  try {
    db = await dbOpen(dbname);
  } catch (e) {
    return new Future.value(null); // dataset does not exist
  }
  if (db == null) return new Future.value(null);

  Object value;
  Completer<List<Float64List>> cpl = new Completer();
  List<Float64List> matrix;
  int nblocks = 0; // counts blocks of rows stored in one piece
  int rowsPerBlock = 0; // counts the rows in a block
  int subrow_length = 0; // >0 if a row gets split before storing
  String json1 = await propGet(db, dsc, DSetStore.BLOCKS_OF_ROWS_INFO);
  if (json1 != null) {
    List<int> blockinfo = UtilsJson.decodeLI(json1);
    nblocks = blockinfo[0];
    rowsPerBlock = blockinfo[1];
    subrow_length = blockinfo[2];
  }

  if (nblocks == 0) {
    // data stored in a single piece
    value = await matrixLoad(db, dsc, dsc);
    if (value is List<Float64List>) {
      matrix = value;
    }
  } else if (subrow_length > 0) {
    // data stored in nblocks subrows. rowsPerBlock subrows form a row.
    matrix = new List<Float64List>();
    List<Float64List> partialRows = new List<Float64List>(rowsPerBlock);
    int partialRowCount = 0;
    for (int i = 0; i < nblocks; i++) {
      String ext = "$BLK_SEP$i";
      // the result is 1 subrow
      value = await matrixLoad(db, dsc, dsc + ext);
      if (value is List<Float64List>) {
        partialRows[partialRowCount] = value[0]; // sub rows is its own block
      }
      partialRowCount++;

      if (partialRowCount == rowsPerBlock) {
        // build row from subrows
        Float64List row = Array2D.join(partialRows);
        matrix.add(row);
        partialRows = new List<Float64List>(rowsPerBlock);
        partialRowCount = 0;
      }
    }
  } else {
    // data stored in nblocks rows
    matrix = new List<Float64List>();
    List<Float64List> rows;
    for (int i = 0; i < nblocks; i++) {
      String ext = "$BLK_SEP$i";
      // the result is a list of complete rows
      value = await matrixLoad(db, dsc, dsc + ext);
      if (value is List<Float64List>) {
        rows = value;
      }
      Array2D.appendRows(rows, matrix);
    }
  }

  cpl.complete(matrix);
  db.close();
  return cpl.future;
}
