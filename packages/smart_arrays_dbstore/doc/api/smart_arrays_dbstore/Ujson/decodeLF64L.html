List<Float64List> decodeLF64L (String jsn)

jsn to be encoded as List

Implementation

static List<Float64List> decodeLF64L(String jsn) {
  dynamic res = json.decode(jsn);

  List<Float64List> levels = new List<Float64List>(res.length);
  for (int k = 0; k < res.length; k++) {
    if (res[k] == null)
      levels[k] = null;
    else {
      levels[k] = Float64List(res[k].length);
      for (int i = 0; i < levels[k].length; i++) levels[k][i] = res[k][i];
    }
  }
  return levels;
}
